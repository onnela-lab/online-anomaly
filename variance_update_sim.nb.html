library(MASS)
eps_func = function(eps){
  m = length(eps)
  qnorm(rank(eps)/(m+1))
}

simdata = function(dist,indep,n,p){
  if(dist=="gaussian" & indep==1){
    gendata = matrix(rnorm(n*p),nrow=n,ncol=p)
  }
  if(dist=="gaussian" & indep==0){
    Sigma = matrix(.7, nrow=p, ncol=p) + diag(p)*.3
    gendata = mvrnorm(n=n, mu=rep(0,p), Sigma=Sigma)
  }
  if(dist=="gamma" & indep==1){
    gendata = matrix(rgamma(n*p,shape=2,scale=2),nrow=n,ncol=p)
  }
  if(dist=="gamma" & indep==0){
    Sigma = matrix(.7, nrow=p, ncol=p) + diag(p)*.3
    temp = mvrnorm(n=n, mu=rep(0,p), Sigma=Sigma)
    gendata = qgamma(pnorm(temp),shape=2,scale=2)
  }
  gendata
}
runsim = function(data,test_index){
  norm1=c();norm2=c();norm3=c();norm4=c()
  stat1=c();stat2=c();stat3=c();stat4=c()
  ranked_eps = data[1:14,]
  for(k in 1:ncol(data)){
    ranked_eps[,k]=eps_func(ranked_eps[,k])
  }
  sigma3 = t(ranked_eps)%*%ranked_eps/14
  sigma4 = cov2cor(sigma3)
  
  ranked_eps = data
    for(k in 1:ncol(data)){
      ranked_eps[,k]=eps_func(ranked_eps[,k])
    }
  best_sigma = cov2cor(t(ranked_eps)%*%ranked_eps/nrow(ranked_eps))
  
  for(i in 15:nrow(data)){
    ranked_eps = data[1:i,]
    for(k in 1:ncol(data)){
      ranked_eps[,k]=eps_func(ranked_eps[,k])
    }
    sigma1 = t(ranked_eps)%*%ranked_eps/i
    sigma2 = cov2cor(sigma1)
    best_sigma = sigma2
    sigma3 = (i-1)/i*sigma3+1/i*outer(ranked_eps[i,],ranked_eps[i,])
    sigma4 = cov2cor((i-1)/i*sigma4+1/i*outer(ranked_eps[i,],ranked_eps[i,]))
    norm1 = c(norm1,norm(sigma1-best_sigma,type='F'))
    norm2 = c(norm2,norm(sigma2-best_sigma,type='F'))
    norm3 = c(norm3,norm(sigma3-best_sigma,type='F'))
    norm4 = c(norm4,norm(sigma4-best_sigma,type='F'))
    if(is.numeric(test_index)==T){
      test = ranked_eps[test_index,]
    }
    if(test_index=="new_obs"){
      test = ranked_eps[i,]
    }
    stat1 = c(stat1,test%*%solve(sigma1)%*%test)
    stat2 = c(stat2,test%*%solve(sigma2)%*%test)
    stat3 = c(stat3,test%*%solve(sigma3)%*%test)
    stat4 = c(stat4,test%*%solve(sigma4)%*%test)
  }
  out = rbind(norm1,norm2,norm3,norm4,stat1,stat2,stat3,stat4)
  out
}
eval_sim = function(out,p){
  par(mfrow=c(1,2),mar=c(5,2,3,2),cex=0.6)
  n = ncol(out)
  scale_min = min(out[1:4,])
  scale_max = max(out[1:4,])
  plot(15:(n+14),out[1,],type = 'l',main = "frobenius norm",xlab="day",ylim=c(scale_min*0.95,scale_max*1.05))
  lines(15:(n+14),out[2,],col="red")
  lines(15:(n+14),out[3,],col="blue")
  lines(15:(n+14),out[4,],col="green")
  legend("topright", legend=c("off_raw","off_scale","online_raw","online_scale"), box.col="white",
         fill=c("black","red","blue","green"),inset=0.01, xpd=TRUE)
  
  scale_min = min(out[5:8,])
  scale_max = max(out[5:8,])
  plot(15:(n+14),out[5,],type = 'l',main = "test statistic",xlab="day",ylim=c(scale_min*0.95,scale_max*1.05))
  lines(15:(n+14),out[6,],col="red")
  lines(15:(n+14),out[7,],col="blue")
  lines(15:(n+14),out[8,],col="green")
  abline(h = qchisq(0.95,df=p))
  legend("topright", legend=c("off_raw","off_scale","online_raw","online_scale"), box.col="white",
         fill=c("black","red","blue","green"),inset=0.01, xpd=TRUE)
  print(mean((out[6,]-out[8,])[100:n]))
}

find_knot = function(tol,input,df){
  dens = density(dat,from=0,n=1000)
  for (j in 100:990){
    x_axis = dens$x[j:(j+10)]

    y_theo = dchisq(x_axis,df)
    if(mean(abs(y_emp-y_theo))>tol){
      break
    }
  }
  pchisq(x_axis[1],df)
}

for(k in 30:150){
  mapping = function(input,df){
    percentile = 0.4-0.03*df
    diff = input - qchisq(percentile,df=df)
    diff[diff<0]=0
    output = input*exp(diff/k)
    output
  }
  error_row = k
  for(df in 4:12){
    dat = mylist[[df-3]][4,]
    new_dat = mapping(dat,df)
    error = sum(new_dat>qchisq(0.95,df=df))/10000
    error_row = c(error_row,error)
  }

mapping = function(input,df){
  diff = input - qchisq(0.4-0.03*df,df=df)
  diff[diff<0]=0
  output = input*exp(diff/(13+7*df))
  output
}
